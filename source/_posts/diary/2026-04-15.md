class Solution {
    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> path = new ArrayList<>();
        boolean[] used = new boolean[nums.length+1];
        back(res, path, used, nums);
        return res;
    }
    private void back(List<List<Integer>> res, List<Integer> path, boolean[] used, int[] nums){
        // 终止条件
        if(path.size() == nums.length){
            res.add(new ArrayList<>(path));
            return;
        }
        for(int i = 0; i < nums.length; i++){
            // 剪枝操作
            if(used[i]){
                continue;
            }
            used[i] = true;
            path.add(nums[i]);
            // 递归
            back(res, path, used, nums);
            // 回溯
            path.remove(path.size()-1);
            used[i] = false;            
        }
    }
}

class Solution {
    public int fib(int n) {
        int a = 0, b = 1;
        int sum = a + b;
        for(int i = 2; i < n; i++){
            a = b;
            b = sum;
            sum = (a+b) % 1000000-7;
        }
        return sum;
    }
}
